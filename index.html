    <button onclick="clockOut()">Clock Out</button>


    <script>

        function sendData(action) {

            let name = document.getElementById("employeeName").value;

            if (!name) {

                alert("Please enter your name.");

                return;

            }

            let url = "https://script.google.com/macros/s/AKfycbydgz3ytDizv9-O7aVCOx8X9rfd8VrNn0hT0Jxc_TbMZulVZ22Z738L-MhJ1jz1Ft3EvA/exec";

            

            fetch(url + "?name=" + encodeURIComponent(name) + "&action=" + action)

                .then(response => response.text())

                .then(data => alert(data))

                .catch(error => alert("Error: " + error));

        }


        function clockIn() { sendData("clockin"); }

        function clockOut() { sendData("clockout"); }

    </script>
